// Function declarations
int generateAddition(int min, int max);
int generateSubtraction(int min, int max);
int generateMultiplication(int min, int max);
int generateDivision(int min, int max);
bool checkAnswer(int correctAnswer, int userAnswer);
int getRandomNumber(int min, int max);

// Main function changed for integration
void generateQuestionSet()
{
    srand(time(0));

    int difficulty;
    int min, max;
    int choice;
    int correctAnswer, userAnswer;
    int score = 0;

    cout << "===== MATH QUIZ GAME =====" << endl;

    cout << "\nSelect Difficulty Level:" << endl;
    cout << "1. Easy (1-10)" << endl;
    cout << "2. Medium (10-50)" << endl;
    cout << "3. Hard (50-100)" << endl;
    cout << "Enter choice: ";
    cin >> difficulty;

    switch (difficulty)
    {
        case 1:
            min = 1;
            max = 10;
            break;

        case 2:
            min = 10;
            max = 50;
            break;

        case 3:
            min = 50;
            max = 100;
            break;

        default:
            cout << "Invalid choice!" << endl;
            return;
    }

    for (int i = 1; i <= 5; i++)
    {
        cout << "\nQuestion " << i << endl;

        cout << "Choose Operation:" << endl;
        cout << "1. Addition" << endl;
        cout << "2. Subtraction" << endl;
        cout << "3. Multiplication" << endl;
        cout << "4. Division" << endl;
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice)
        {
            case 1:
                correctAnswer = generateAddition(min, max);
                break;

            case 2:
                correctAnswer = generateSubtraction(min, max);
                break;

            case 3:
                correctAnswer = generateMultiplication(min, max);
                break;

            case 4:
                correctAnswer = generateDivision(min, max);
                break;

            default:
                cout << "Invalid operation!" << endl;
                i--;
                continue;
        }

        cout << "Enter your answer: ";
        cin >> userAnswer;

        if (checkAnswer(correctAnswer, userAnswer))
        {
            cout << "Correct Answer!" << endl;
            score++;
        }
        else
        {
            cout << "Wrong Answer!" << endl;
            cout << "Correct Answer is: " << correctAnswer << endl;
        }
    }

    cout << "\n===== QUIZ COMPLETED =====" << endl;
    cout << "Your Score: " << score << "/5" << endl;
}

// Random number generator
int getRandomNumber(int min, int max)
{
    return rand() % (max - min + 1) + min;
}

// Addition
int generateAddition(int min, int max)
{
    int a = getRandomNumber(min, max);
    int b = getRandomNumber(min, max);

    cout << a << " + " << b << " = ";
    return a + b;
}

// Subtraction
int generateSubtraction(int min, int max)
{
    int a = getRandomNumber(min, max);
    int b = getRandomNumber(min, max);

    cout << a << " - " << b << " = ";
    return a - b;
}

// Multiplication
int generateMultiplication(int min, int max)
{
    int a = getRandomNumber(min, max);
    int b = getRandomNumber(min, max);

    cout << a << " * " << b << " = ";
    return a * b;
}

// Division
int generateDivision(int min, int max)
{
    int b = getRandomNumber(min, max);
    int answer = getRandomNumber(min, max);

    int a = b * answer;

    cout << a << " / " << b << " = ";
    return answer;
}

// Answer checking
bool checkAnswer(int correctAnswer, int userAnswer)
{
    return correctAnswer == userAnswer;
}
